class Solution {
public:
    int longestValidParentheses(string s) {
        stack<int> a;
        int ans = 0;
        short ch[20000]={ 0 };

        //bool * ch = new bool[s.size()]{ 0 };
        for (int i = 0; i<s.size(); i++) {
            if (a.empty()) a.push(s[i]);
            else {
                if (a.top() == '('&&s[i] == ')') {
                    ch[i] = 1;
                    a.pop();
                }
                else {
                    a.push(s[i]);
                }
            }
        }
        for (int i = s.size() - 1; i >= 0; i--) {
            int ind = i - 1;
            if (ch[i] == 1) {
                while (ch[ind] != 0)ind--;
                ch[ind] = 2;
            }
        }
        int maxx = 0;
        int i = 0;
        int tem = 0;
        while (i < s.size()) {
            tem = 0;
            while (ch[i] != 0) i++,tem++;
            i++;
            maxx = tem > maxx ? tem : maxx;
        }
        return maxx;
    }
};
