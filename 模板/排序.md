	for (int i = 0; i < n; i++) {
		for (int j = 1; j < n-i; j++)

快速：  
```cpp
void qs(int st, int ed) {

	
	int i = st, j = ed, mid, temp;
	mid = ar[(i + j) / 2];

	while (i < j) {
		while (ar[i] < mid) i++;
		while (ar[j] > mid) j--;//这里两行都不能有等号，防止i或者j越过mid，可能导致数组最终越界，i大于n或者j小于0

		if (i <= j) {
			
			temp = ar[i];
			ar[i] = ar[j];
			ar[j] = temp;
			i++;
			j--;
		}
		
	}


	if (st < j) qs(st, j);
	if (ed > i) qs(i, ed);
}
```
