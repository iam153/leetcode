#include<bits/stdc++.h>
using namespace std;

const int maxn = 2e3 + 5;
int gra[maxn][maxn] = { 0 }, wei[maxn] = { 0 };
map<string, int> s2i;
map<int, string> i2s;
map<string, int> gang;
bool vis[maxn] = { 0 };
int n, k, numP;


void dfss(int u, int &head, int &nums, int &val) {
	vis[u] = 1;
	nums++;
	if (wei[u] > wei[head])
		head = u;
	for(int i = 0; i < numP; i++)
		if (gra[u][i] > 0) {
			val += gra[u][i];
			gra[u][i] = gra[i][u] = 0;
			if (!vis[i])
				dfss(i, head, nums, val);
		}
}

void dfs() {
	for (int i = 0; i < numP; i++) {
		if (!vis[i]) {
			int head = i, nums = 0, val = 0;
			dfss(i, head, nums, val);
			if (val > k && nums > 2)
				gang[i2s[head]] = nums;
		}
	}
}


int main() {
	string s1, s2;
	int w;
	cin >> n >> k;
	numP = 0;
	while (n--) {
		cin >> s1 >> s2 >> w;
		if (!s2i.count(s1)) {
			s2i[s1] = numP;
			i2s[numP] = s1;
			numP++;
		}
		if (!s2i.count(s2)) {
			s2i[s2] = numP;
			i2s[numP] = s2;
			numP++;
		}
		int id1 = s2i[s1], id2 = s2i[s2];
		wei[id1] += w;
		wei[id2] += w;
		gra[id1][id2] += w;
		gra[id2][id1] += w;
	}

	dfs();

	cout << gang.size() << endl;
	for (auto it = gang.begin(); it != gang.end(); it++)
		cout << it->first << " " << it->second << endl;
	return 0;
}
