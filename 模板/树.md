```cpp
	if (postL > postR)
		return NULL;
	node* root = new node;
	root->val = post[postR];
	int k;
	for (k = inL; k <= inR; k++)
		if (in[k] == post[postR])
			break;
	int nL = k - inL;
	root->left = create(inL, k - 1, postL, postL + nL - 1);
	root->right = create(k + 1, inR, postL + nL, postR - 1);
```

```cpp
node* create(int preL, int preR, int inL, int inR) {
	if (preL > preR)
		return NULL;

	node* rt = new node;
	rt->val = pre[preL];
	int k;
	for (k = inL; k < inR; k++)
		if (in[k] == pre[preL])
			break;
	int nL = k - inL;
	rt->le = create(preL + 1, preL + nL, inL, k - 1);
	rt->ri = create(preL + nL + 1, preR, k + 1, inR);
	return rt;
}
```

```cpp
//avl
struct node {
	int hig, val;
	node* le, *ri;
};

node* init(int v) {
	node* t = new node;
	t->val = v;
	t->le = t->ri = NULL;
	t->hig = 1;
	return t;
}

int gh(node* rt) {
	if (rt == NULL)
		return 0;
	return rt->hig;
}

int gb(node* rt) {
	return gh(rt->le) - gh(rt->ri);
}

void uh(node* rt) {
	rt->hig = max(gh(rt->le), gh(rt->ri)) + 1;
}

void L(node* &rt) {

	node* tem = rt->ri;
	rt->ri = tem->le;
	tem->le = rt;
	uh(rt);
	uh(tem);
	rt = tem;
}

void R(node* &rt) {

	node* tem = rt->le;
	rt->le = tem->ri;
	tem->ri = rt;
	uh(rt);
	uh(tem);
	rt = tem;
}

void ins(node* &rt, int v) {
	if (rt == NULL) {
		rt = init(v);
		return;
	}
	if (v < rt->val) {
		ins(rt->le, v);
		uh(rt);
		if (gb(rt) == 2) {
			if (gb(rt->le) == 1) {
				R(rt);
			}
			else if (gb(rt->le) == -1) {
				L(rt->le);
				R(rt);
			}
		}
	}
	else {
		ins(rt->ri, v);
		uh(rt);
		if (gb(rt) == -2) {
			if (gb(rt->ri) == -1) {
				L(rt);
			}
			else if (gb(rt->ri) == 1) {
				R(rt->ri);
				L(rt);
			}
		}
	}
}
```
