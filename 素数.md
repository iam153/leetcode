辗转相除  

素数筛  
1、普通遍历(n为顶)、sqrt(n)为顶  
2、埃氏筛， j从2i开始，j从i^2开始  
3、欧拉筛，外壳i感觉可以上顶到n/2 + 1，这样可以把<=n的质数和合数都筛完

以下为代码
```cpp
#include<iostream>
#include<cmath>
using namespace std;

#define maxn int(1e4)


bool mark[maxn] = { 1,1 };//1表示合数,先把0和1置为1，实际上0两者都不是
int prime[maxn];//存质数

void brutal() {
	for (int i = 2; i <= maxn; i++) 
		for (int j = 2; j < i; j++)
			if (i%j == 0)
				mark[i] = 1;
	
}

void sqrBru() {
	for (int i = 2; i <= maxn; i++)
		for (int j = 2; j <= sqrt(i); j++)
			if (i%j == 0)
				mark[i] = 1;
}

void Era() {
	for (int i = 2; i <= maxn; i++) 
		if (mark[i] == 0)
			for (int j = 2 * i; j <= maxn; j += i)
				mark[j] = 1;
	
}

void Era_2() {
	for (int i = 2; i <= maxn; i++)
		if (mark[i] == 0)
			for (int j = i * i; j <= maxn; j += i)//i*i可能溢出
				mark[j] = 1;
}

void euler() {
	int ind = 1;
	for (int i = 2; i <= maxn; i++) {//上顶可以改成i<maxn/2+1
		if (mark[i] == 0) {
			prime[ind] = i;
			ind++;
		}
		for (int j = 1; j < ind&&i*prime[j] <= maxn; j++) {
			mark[i*prime[j]] = 1;
			if (i%prime[j] == 0)
				break;
		}

	}
}

int main() {
	brutal();
	bool m1[maxn];
	for (int i = 0; i < maxn; i++)
		m1[i] = mark[i];

	sqrBru();
	bool m2[maxn];
	for (int i = 0; i < maxn; i++)
		m2[i] = mark[i];

	Era();
	bool m3[maxn];
	for (int i = 0; i < maxn; i++)
		m3[i] = mark[i];

	Era_2();
	bool m4[maxn];
	for (int i = 0; i < maxn; i++)
		m4[i] = mark[i];

	euler();
	bool m5[maxn];
	for (int i = 0; i < maxn; i++)
		m5[i] = mark[i];

	for (int i = 0; i < maxn; i++)
		if (m1[i] != m2[i] || m1[i] != m3[i] || m1[i] != m4[i] || m1[i] != m5[i])
			cout << "i: " << i <<" "<< m1[i] << " " << m2[i] << " " << m3[i] << " " << m4[i] << " " << m5[i] << endl;
	cout << "end" << endl;
	system("pause");
}
```
